# 鼠标交互处理

## Canvas 坐标系统

### 问题
Canvas 坐标与屏幕坐标不一致（缩放、边距、高分屏）。

### 解决方案
```javascript
function getCanvasPosition(event, canvas) {
  const rect = canvas.getBoundingClientRect();
  const scaleX = canvas.width / rect.width;
  const scaleY = canvas.height / rect.height;
  
  return {
    x: (event.clientX - rect.left) * scaleX,
    y: (event.clientY - rect.top) * scaleY
  };
}

// 使用
canvas.addEventListener('mousemove', (e) => {
  const pos = getCanvasPosition(e, canvas);
  console.log('Canvas position:', pos.x, pos.y);
});
```

## 基础鼠标事件

### 点击检测
```javascript
canvas.addEventListener('click', (e) => {
  const pos = getCanvasPosition(e, canvas);
  
  // 检测点击了哪个对象
  const clickedObject = objects.find(obj => isPointInObject(pos, obj));
  
  if (clickedObject) {
    handleClick(clickedObject);
  }
});

function isPointInObject(point, obj) {
  if (obj.type === 'circle') {
    const dx = point.x - obj.x;
    const dy = point.y - obj.y;
    return Math.sqrt(dx * dx + dy * dy) <= obj.radius;
  }
  
  if (obj.type === 'rect') {
    return point.x >= obj.x && point.x <= obj.x + obj.width &&
           point.y >= obj.y && point.y <= obj.y + obj.height;
  }
  
  return false;
}
```

### 悬停效果
```javascript
let hoveredObject = null;

canvas.addEventListener('mousemove', (e) => {
  const pos = getCanvasPosition(e, canvas);
  
  const newHovered = objects.find(obj => isPointInObject(pos, obj));
  
  if (newHovered !== hoveredObject) {
    hoveredObject = newHovered;
    canvas.style.cursor = hoveredObject ? 'pointer' : 'default';
    render(); // 重绘以显示悬停效果
  }
});

function render() {
  ctx.clearRect(0, 0, width, height);
  
  objects.forEach(obj => {
    const isHovered = obj === hoveredObject;
    obj.render(ctx, isHovered);
  });
}

// 对象渲染时根据悬停状态改变样式
Circle.prototype.render = function(ctx, isHovered) {
  ctx.beginPath();
  ctx.fillStyle = isHovered ? this.hoverColor : this.color;
  ctx.arc(this.x, this.y, isHovered ? this.radius * 1.1 : this.radius, 0, Math.PI * 2);
  ctx.fill();
};
```

## 拖拽交互

### 基础拖拽
```javascript
let isDragging = false;
let dragObject = null;
let dragOffset = { x: 0, y: 0 };

canvas.addEventListener('mousedown', (e) => {
  const pos = getCanvasPosition(e, canvas);
  
  // 从上到下检测（先渲染的在上层）
  for (let i = objects.length - 1; i >= 0; i--) {
    if (isPointInObject(pos, objects[i])) {
      isDragging = true;
      dragObject = objects[i];
      dragOffset.x = pos.x - dragObject.x;
      dragOffset.y = pos.y - dragObject.y;
      break;
    }
  }
});

canvas.addEventListener('mousemove', (e) => {
  if (!isDragging || !dragObject) return;
  
  const pos = getCanvasPosition(e, canvas);
  dragObject.x = pos.x - dragOffset.x;
  dragObject.y = pos.y - dragOffset.y;
  
  render();
});

canvas.addEventListener('mouseup', () => {
  isDragging = false;
  dragObject = null;
});

canvas.addEventListener('mouseleave', () => {
  isDragging = false;
  dragObject = null;
});
```

### 拖拽优化（大量对象）
```javascript
// 使用空间索引加速命中检测
class SpatialHash {
  constructor(cellSize) {
    this.cellSize = cellSize;
    this.cells = new Map();
  }
  
  clear() {
    this.cells.clear();
  }
  
  insert(obj) {
    const minX = Math.floor(obj.x / this.cellSize);
    const maxX = Math.floor((obj.x + obj.width) / this.cellSize);
    const minY = Math.floor(obj.y / this.cellSize);
    const maxY = Math.floor((obj.y + obj.height) / this.cellSize);
    
    for (let x = minX; x <= maxX; x++) {
      for (let y = minY; y <= maxY; y++) {
        const key = `${x},${y}`;
        if (!this.cells.has(key)) {
          this.cells.set(key, []);
        }
        this.cells.get(key).push(obj);
      }
    }
  }
  
  query(x, y) {
    const key = `${Math.floor(x / this.cellSize)},${Math.floor(y / this.cellSize)}`;
    return this.cells.get(key) || [];
  }
}

const spatialHash = new SpatialHash(100);

function getObjectsAt(x, y) {
  const candidates = spatialHash.query(x, y);
  return candidates.filter(obj => isPointInObject({ x, y }, obj));
}

// 每帧更新空间索引
function render() {
  spatialHash.clear();
  objects.forEach(obj => spatialHash.insert(obj));
  
  // 渲染...
}
```

## 矩形选择（框选）

### 实现
```javascript
let isSelecting = false;
let selectionStart = { x: 0, y: 0 };
let selectionEnd = { x: 0, y: 0 };
let selectedObjects = [];

canvas.addEventListener('mousedown', (e) => {
  if (e.shiftKey) { // Shift + 点击开始框选
    isSelecting = true;
    const pos = getCanvasPosition(e, canvas);
    selectionStart = pos;
    selectionEnd = pos;
    selectedObjects = [];
  }
});

canvas.addEventListener('mousemove', (e) => {
  if (!isSelecting) return;
  
  selectionEnd = getCanvasPosition(e, canvas);
  render();
  
  // 绘制选择框
  ctx.save();
  ctx.strokeStyle = '#0066ff';
  ctx.fillStyle = 'rgba(0, 102, 255, 0.1)';
  ctx.lineWidth = 1;
  ctx.setLineDash([5, 5]);
  
  const x = Math.min(selectionStart.x, selectionEnd.x);
  const y = Math.min(selectionStart.y, selectionEnd.y);
  const width = Math.abs(selectionEnd.x - selectionStart.x);
  const height = Math.abs(selectionEnd.y - selectionStart.y);
  
  ctx.fillRect(x, y, width, height);
  ctx.strokeRect(x, y, width, height);
  ctx.restore();
});

canvas.addEventListener('mouseup', () => {
  if (!isSelecting) return;
  
  // 检测框内对象
  const x = Math.min(selectionStart.x, selectionEnd.x);
  const y = Math.min(selectionStart.y, selectionEnd.y);
  const width = Math.abs(selectionEnd.x - selectionStart.x);
  const height = Math.abs(selectionEnd.y - selectionStart.y);
  
  selectedObjects = objects.filter(obj => isRectIntersect(obj, { x, y, width, height }));
  
  isSelecting = false;
  render();
});

function isRectIntersect(obj, rect) {
  return obj.x >= rect.x &&
         obj.y >= rect.y &&
         obj.x + obj.width <= rect.x + rect.width &&
         obj.y + obj.height <= rect.y + rect.height;
}
```

## 鼠标滚轮缩放

### 画布缩放
```javascript
let viewport = {
  x: 0,
  y: 0,
  zoom: 1
};

canvas.addEventListener('wheel', (e) => {
  e.preventDefault();
  
  const pos = getCanvasPosition(e, canvas);
  const zoomFactor = e.deltaY > 0 ? 0.9 : 1.1;
  
  // 以鼠标位置为中心缩放
  const worldX = (pos.x - viewport.x) / viewport.zoom;
  const worldY = (pos.y - viewport.y) / viewport.zoom;
  
  viewport.zoom *= zoomFactor;
  viewport.zoom = Math.max(0.1, Math.min(10, viewport.zoom)); // 限制缩放范围
  
  viewport.x = pos.x - worldX * viewport.zoom;
  viewport.y = pos.y - worldY * viewport.zoom;
  
  render();
});

function render() {
  ctx.clearRect(0, 0, width, height);
  ctx.save();
  
  // 应用视口变换
  ctx.translate(viewport.x, viewport.y);
  ctx.scale(viewport.zoom, viewport.zoom);
  
  // 绘制世界坐标中的对象
  objects.forEach(obj => obj.render(ctx));
  
  ctx.restore();
  
  // 绘制 UI 覆盖层（不受视口影响）
  renderUI();
}
```

## 像素级命中检测

### 非矩形对象检测
```javascript
function isPointInShape(x, y, shape) {
  // 方法1：路径检测
  if (shape.path) {
    ctx.beginPath();
    shape.path.forEach(cmd => {
      if (cmd.type === 'moveTo') ctx.moveTo(cmd.x, cmd.y);
      if (cmd.type === 'lineTo') ctx.lineTo(cmd.x, cmd.y);
      if (cmd.type === 'arc') ctx.arc(cmd.x, cmd.y, cmd.radius, cmd.startAngle, cmd.endAngle);
    });
    ctx.closePath();
    return ctx.isPointInPath(x, y);
  }
  
  // 方法2：像素级检测
  if (shape.imageData) {
    const pixelX = Math.floor(x - shape.x);
    const pixelY = Math.floor(y - shape.y);
    
    if (pixelX < 0 || pixelX >= shape.width || pixelY < 0 || pixelY >= shape.height) {
      return false;
    }
    
    const index = (pixelY * shape.width + pixelX) * 4;
    const alpha = shape.imageData.data[index + 3];
    return alpha > 0;
  }
  
  return false;
}
```

## 手势识别

### 简单手势
```javascript
class GestureDetector {
  constructor(canvas) {
    this.canvas = canvas;
    this.points = [];
    this.isTracking = false;
    
    this.setupListeners();
  }
  
  setupListeners() {
    this.canvas.addEventListener('mousedown', (e) => {
      this.isTracking = true;
      this.points = [getCanvasPosition(e, this.canvas)];
    });
    
    this.canvas.addEventListener('mousemove', (e) => {
      if (!this.isTracking) return;
      this.points.push(getCanvasPosition(e, this.canvas));
    });
    
    this.canvas.addEventListener('mouseup', () => {
      if (!this.isTracking) return;
      this.isTracking = false;
      
      const gesture = this.recognizeGesture(this.points);
      if (gesture) {
        this.onGesture(gesture);
      }
      this.points = [];
    });
  }
  
  recognizeGesture(points) {
    if (points.length < 10) return null;
    
    // 简单示例：检测圆形
    const first = points[0];
    const last = points[points.length - 1];
    const distance = Math.sqrt(
      Math.pow(last.x - first.x, 2) + Math.pow(last.y - first.y, 2)
    );
    
    if (distance < 20) {
      return 'circle';
    }
    
    // 可以添加更多手势识别逻辑
    return null;
  }
  
  onGesture(gesture) {
    console.log('Gesture detected:', gesture);
  }
}
```

```javascript
const InteractionState = {
  IDLE: 'idle',
  DRAGGING: 'dragging',
  PANNING: 'panning',
  SELECTING: 'selecting',
  DRAWING: 'drawing'
};

class InteractionManager {
  constructor(canvas) {
    this.canvas = canvas;
    this.state = InteractionState.IDLE;
    this.setupListeners();
  }
  
  setupListeners() {
    this.canvas.addEventListener('mousedown', (e) => this.handleMouseDown(e));
    this.canvas.addEventListener('mousemove', (e) => this.handleMouseMove(e));
    this.canvas.addEventListener('mouseup', (e) => this.handleMouseUp(e));
  }
  
  handleMouseDown(e) {
    switch (this.state) {
      case InteractionState.IDLE:
        if (e.button === 1) {
          this.state = InteractionState.PANNING;
        } else if (e.shiftKey) {
          this.state = InteractionState.SELECTING;
        } else {
          const obj = this.getObjectAt(e);
          if (obj) {
            this.state = InteractionState.DRAGGING;
            this.startDrag(obj, e);
          }
        }
        break;
    }
  }
  
  handleMouseMove(e) {
    switch (this.state) {
      case InteractionState.DRAGGING:
        this.updateDrag(e);

      case InteractionState.PANNING:
        this.updatePan(e);
        break;
      case InteractionState.SELECTING:
        this.updateSelection(e);
        break;

  }
  
  handleMouseUp(e) {
    this.state = InteractionState.IDLE;
  }
```

## 性能优化

### 防抖与节流
```javascript
// 节流：限制执行频率
function throttle(func, limit) {
  let inThrottle = false;
  return function(...args) {
    if (!inThrottle) {
      func.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}

// 防抖：延迟执行直到停止触发
function debounce(func, delay) {
  let timeoutId;
  return function(...args) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => func.apply(this, args), delay);
  };
}

// 使用
canvas.addEventListener('mousemove', throttle(handleMouseMove, 16)); // ~60fps
```

### 按需渲染
```javascript
let needsRender = false;

function requestRender() {
  if (!needsRender) {
    needsRender = true;
    requestAnimationFrame(() => {
      render();
      needsRender = false;
    });
  }
}

// 交互时请求渲染
canvas.addEventListener('mousemove', () => {
```
