```javascript
function handleTouchStart(e) {

function handleTouchMove(e) {
  e.preventDefault();
  
  // e.touches - 当前所有触摸点
  // e.changedTouches - 本次变化的触摸点
  // e.targetTouches - 当前目标元素上的触摸点
  
  for (let i = 0; i < e.changedTouches.length; i++) {
    const touch = e.changedTouches[i];
    const pos = getCanvasPosition(touch, canvas);
    handleTouchMoveAt(touch.identifier, pos);
  }
}

function handleTouchEnd(e) {
  for (let i = 0; i < e.changedTouches.length; i++) {
    const touch = e.changedTouches[i];
    handleTouchEndAt(touch.identifier);
  }
}
```

### 坐标转换（统一鼠标和触摸）
```javascript
function getCanvasPosition(event, canvas) {
  const rect = canvas.getBoundingClientRect();
  const scaleX = canvas.width / rect.width;
  const scaleY = canvas.height / rect.height;
  
  let clientX, clientY;
  
  if (event.touches && event.touches.length > 0) {
    clientX = event.touches[0].clientX;
    clientY = event.touches[0].clientY;
  } else if (event.changedTouches && event.changedTouches.length > 0) {
    clientX = event.changedTouches[0].clientX;
    clientY = event.changedTouches[0].clientY;
  } else {
    clientX = event.clientX;
    clientY = event.clientY;
  }
  
  return {
    x: (clientX - rect.left) * scaleX,
    y: (clientY - rect.top) * scaleY
  };
}

// 统一事件处理
function handlePointerDown(e) {
  const pos = getCanvasPosition(e, canvas);
  handleInteraction('down', pos, e);
}

function handlePointerMove(e) {
  const pos = getCanvasPosition(e, canvas);
  handleInteraction('move', pos, e);
}

function handlePointerUp(e) {
  handleInteraction('up', null, e);
}

// 绑定两种事件
canvas.addEventListener('mousedown', handlePointerDown);
canvas.addEventListener('touchstart', handlePointerDown, { passive: false });
canvas.addEventListener('mousemove', handlePointerMove);
canvas.addEventListener('touchmove', handlePointerMove, { passive: false });
canvas.addEventListener('mouseup', handlePointerUp);
canvas.addEventListener('touchend', handlePointerUp);
```

## 触摸手势识别

### 单指拖拽
```javascript
let activeTouch = null;
let touchOffset = { x: 0, y: 0 };

function handleTouchStart(e) {
  e.preventDefault();
  
  if (e.touches.length === 1) {
    const touch = e.touches[0];
    const pos = getCanvasPosition(touch, canvas);
    
    const touchedObject = objects.find(obj => isPointInObject(pos, obj));
    
    if (touchedObject) {
      activeTouch = {
        id: touch.identifier,
        object: touchedObject
      };
      touchOffset.x = pos.x - touchedObject.x;
      touchOffset.y = pos.y - touchedObject.y;
    }
  }
}

function handleTouchMove(e) {
  e.preventDefault();
  
  if (!activeTouch) return;
  
  const touch = Array.from(e.touches).find(t => t.identifier === activeTouch.id);
  if (!touch) return;
  
  const pos = getCanvasPosition(touch, canvas);
  activeTouch.object.x = pos.x - touchOffset.x;
  activeTouch.object.y = pos.y - touchOffset.y;
  
  render();
}

function handleTouchEnd(e) {
  for (let i = 0; i < e.changedTouches.length; i++) {
    const touch = e.changedTouches[i];
    if (activeTouch && touch.identifier === activeTouch.id) {
      activeTouch = null;
    }
  }
}
```

### 双指缩放（Pinch Zoom）
```javascript
let pinchStartDistance = 0;
let pinchStartZoom = 1;

function getTouchDistance(touch1, touch2) {
  const dx = touch2.clientX - touch1.clientX;
  const dy = touch2.clientY - touch1.clientY;
  return Math.sqrt(dx * dx + dy * dy);
}

function handleTouchStart(e) {
  e.preventDefault();
  
  if (e.touches.length === 2) {
    pinchStartDistance = getTouchDistance(e.touches[0], e.touches[1]);
    pinchStartZoom = viewport.zoom;
  }
}

function handleTouchMove(e) {
  e.preventDefault();
  
  if (e.touches.length === 2) {
    const currentDistance = getTouchDistance(e.touches[0], e.touches[1]);
    const scale = currentDistance / pinchStartDistance;
    
    viewport.zoom = pinchStartZoom * scale;
    viewport.zoom = Math.max(0.1, Math.min(10, viewport.zoom));
    
    render();
  }
}
```

### 双指旋转
```javascript
let pinchStartAngle = 0;
let pinchStartRotation = 0;

function getTouchAngle(touch1, touch2) {
  return Math.atan2(touch2.clientY - touch1.clientY, touch2.clientX - touch1.clientX);
}

function handleTouchStart(e) {
  e.preventDefault();
  
  if (e.touches.length === 2) {
    pinchStartAngle = getTouchAngle(e.touches[0], e.touches[1]);
    pinchStartRotation = viewport.rotation || 0;
  }
}

function handleTouchMove(e) {
  e.preventDefault();
  
  if (e.touches.length === 2) {
    const currentAngle = getTouchAngle(e.touches[0], e.touches[1]);
    const rotation = currentAngle - pinchStartAngle;
    
    viewport.rotation = pinchStartRotation + rotation;
    render();
  }
}
```

## 移动端适配

### 响应式 Canvas
```javascript
function setupResponsiveCanvas(canvas) {
  function resize() {
    const container = canvas.parentElement;
    const width = container.clientWidth;
    const height = container.clientHeight;
    
    const dpr = window.devicePixelRatio || 1;
    canvas.width = width * dpr;
    canvas.height = height * dpr;
    canvas.style.width = width + 'px';
    canvas.style.height = height + 'px';
    
    const ctx = canvas.getContext('2d');
    ctx.scale(dpr, dpr);
    
    render();
  }
  
  window.addEventListener('resize', resize);
  resize();
}
```

```javascript
function setupMobileOptimization(canvas) {

    // 减少粒子数量

    // 降低渲染帧率

  }
}

// 动态质量调整
let frameCount = 0;
let lastFPSCheck = performance.now();
let currentFPS = 60;

function renderLoop() {
  frameCount++;
  
  const now = performance.now();
  if (now - lastFPSCheck >= 1000) {
    currentFPS = frameCount;
    frameCount = 0;
    lastFPSCheck = now;
    
    // 如果帧率低，降低质量
    if (currentFPS < 30) {
      reduceQuality();
    }
  }
  
  render();
  requestAnimationFrame(renderLoop);
}

function reduceQuality() {
  // 降低分辨率
  renderScale = Math.max(0.5, renderScale - 0.1);
  
  // 减少粒子
  maxParticles = Math.max(100, maxParticles - 50);
}
```

### 多点触控支持
```javascript
const touches = new Map();

function handlePointerDown(e) {
  e.preventDefault();
  canvas.setPointerCapture(e.pointerId);
  
  const pos = getCanvasPosition(e, canvas);
  touches.set(e.pointerId, {
    id: e.pointerId,
    pos: pos,
    startPos: { ...pos },
    type: e.pointerType
  });
  
  handleMultiTouch();
}

function handlePointerMove(e) {
  const touch = touches.get(e.pointerId);
  if (!touch) return;
  
  const pos = getCanvasPosition(e, canvas);
  touch.pos = pos;
  
  handleMultiTouch();
}

function handlePointerUp(e) {
  touches.delete(e.pointerId);
  handleMultiTouch();
}

function handleMultiTouch() {
  if (touches.size === 1) {
    // 单指操作
    const touch = Array.from(touches.values())[0];
    handleSingleTouch(touch);
  } else if (touches.size === 2) {
    // 双指操作
    const [touch1, touch2] = Array.from(touches.values());
    handlePinchGesture(touch1, touch2);
  }
}
```

### 设备能力检测
```javascript
function detectDeviceCapabilities() {
  const isMobile = /Android|iPhone|iPad|iPod|webOS/i.test(navigator.userAgent);
  const isLowEnd = navigator.hardwareConcurrency <= 2;
  const isSlowNetwork = navigator.connection && 
    (navigator.connection.effectiveType === '2g' || 
     navigator.connection.effectiveType === 'slow-2g');
  
  return {
    isMobile,
    isLowEnd,
    isSlowNetwork,
    recommendedSettings: {
      maxParticles: isLowEnd ? 200 : isMobile ? 500 : 2000,
      targetFPS: isLowEnd ? 30 : 60,
      renderScale: isLowEnd ? 0.5 : isMobile ? 0.75 : 1.0,
      maxDPR: isLowEnd ? 1 : isMobile ? 1.5 : 2
    }
  };
}

const capabilities = detectDeviceCapabilities();
applySettings(capabilities.recommendedSettings);
```

## 最佳实践

1. **统一事件处理**：使用 Pointer Events 或封装层
2. **禁用默认手势**：`touch-action: none`
3. **响应式设计**：自适应屏幕尺寸和方向
4. **性能优先**：移动端降低质量保帧率
5. **真实测试**：在真实设备上测试
6. **渐进增强**：根据设备能力调整质量
