# 粒子系统设计模式

## 粒子系统架构

### 基础粒子系统
```javascript
class Particle {
  constructor() {
    this.reset();
  }
  
  reset() {
    this.x = 0;
    this.y = 0;
    this.vx = 0;
    this.vy = 0;
    this.life = 0;
    this.maxLife = 0;
    this.size = 0;
    this.color = '';
    this.active = false;
    return this;
  }
  
  init(x, y, vx, vy, life, size, color) {
    this.x = x;
    this.y = y;
    this.vx = vx;
    this.vy = vy;
    this.life = 0;
    this.maxLife = life;
    this.size = size;
    this.color = color;
    this.active = true;
    return this;
  }
  
  update(dt) {
    if (!this.active) return;
    
    this.life += dt;
    if (this.life >= this.maxLife) {
      this.active = false;
      return;
    }
    
    this.x += this.vx * dt;
    this.y += this.vy * dt;
  }
  
  render(ctx) {
    if (!this.active) return;
    
    const progress = this.life / this.maxLife;
    const alpha = 1 - progress;
    
    ctx.save();
    ctx.globalAlpha = alpha;
    ctx.fillStyle = this.color;
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
    ctx.fill();
    ctx.restore();
  }
}
```

### 粒子发射器（Emitter）
```javascript
class ParticleEmitter {
  constructor(options = {}) {
    this.x = options.x || 0;
    this.y = options.y || 0;
    this.rate = options.rate || 10; // 每秒发射数
    this.burst = options.burst || 0; // 爆发数量
    this.speed = options.speed || 100;
    this.speedVariance = options.speedVariance || 0.5;
    this.angle = options.angle || 0; // 弧度
    this.spread = options.spread || Math.PI * 2; // 扩散角度
    this.life = options.life || 1; // 粒子寿命
    this.lifeVariance = options.lifeVariance || 0.5;
    this.size = options.size || 5;
    this.sizeVariance = options.sizeVariance || 0.5;
    this.color = options.color || '#ffffff';
    this.colors = options.colors || null; // 颜色数组
    this.gravity = options.gravity || 0;
    this.wind = options.wind || 0;
    this.friction = options.friction || 1;
    
    this.particles = [];
    this.emitTimer = 0;
  }
  
  emit() {
    const speed = this.speed * (1 + (Math.random() - 0.5) * this.speedVariance * 2);
    const angle = this.angle + (Math.random() - 0.5) * this.spread;
    const vx = Math.cos(angle) * speed;
    const vy = Math.sin(angle) * speed;
    const life = this.life * (1 + (Math.random() - 0.5) * this.lifeVariance * 2);
    const size = this.size * (1 + (Math.random() - 0.5) * this.sizeVariance * 2);
    const color = this.colors 
      ? this.colors[Math.floor(Math.random() * this.colors.length)]
      : this.color;
    
    const particle = new Particle();
    particle.init(
      this.x, this.y,
      vx, vy,
      Math.max(0.1, life),
      Math.max(1, size),
      color
    );
    
    this.particles.push(particle);
  }
  
  burst(count) {
    for (let i = 0; i < count; i++) {
      this.emit();
    }
  }
  
  update(dt) {
    // 持续发射
    this.emitTimer += dt;
    const emitCount = Math.floor(this.emitTimer * this.rate);
    if (emitCount > 0) {
      for (let i = 0; i < emitCount; i++) {
        this.emit();
      }
      this.emitTimer -= emitCount / this.rate;
    }
    
    // 更新粒子
    for (let i = this.particles.length - 1; i >= 0; i--) {
      const p = this.particles[i];
      
      if (!p.active) {
        this.particles.splice(i, 1);
        continue;
      }
      
      p.vy += this.gravity * dt;
      p.vx += this.wind * dt;
      p.vx *= this.friction;
      p.vy *= this.friction;
      
      p.update(dt);
    }
  }
  
  render(ctx) {
    for (const p of this.particles) {
      p.render(ctx);
    }
  }
}
```

```javascript
      y: Math.random() * height,

  return {
    update(dt, time) {
      // 星星位置固定，只需更新闪烁
    },

      for (const star of stars) {
        const twinkle = Math.sin(time * star.twinkleSpeed + star.twinkleOffset);
        const alpha = star.brightness * (0.7 + 0.3 * twinkle);
        
        ctx.save();
        ctx.globalAlpha = alpha;
        ctx.fillStyle = '#ffffff';
        ctx.beginPath();
        ctx.arc(star.x, star.y, star.size, 0, Math.PI * 2);
        ctx.fill();
        ctx.restore();
      }
    }
  };
}
```

## 高级技巧

### 1. 粒子间交互
```javascript
function applyParticleInteractions(particles, radius = 50) {
  // 简单斥力模拟
  for (let i = 0; i < particles.length; i++) {
    for (let j = i + 1; j < particles.length; j++) {
      const p1 = particles[i];
      const p2 = particles[j];
      
      if (!p1.active || !p2.active) continue;
      
      const dx = p2.x - p1.x;
      const dy = p2.y - p1.y;
      const dist = Math.sqrt(dx * dx + dy * dy);
      
      if (dist < radius && dist > 0) {
        const force = (radius - dist) / radius * 0.5;
        const fx = (dx / dist) * force;
        const fy = (dy / dist) * force;
        
        p1.vx -= fx;
        p1.vy -= fy;
        p2.vx += fx;
        p2.vy += fy;
      }
    }
  }
}
```

### 2. 空间分区优化
```javascript
class SpatialGrid {
  constructor(cellSize) {
    this.cellSize = cellSize;
    this.cells = new Map();
  }
  
  clear() {
    this.cells.clear();
  }
  
  getKey(x, y) {
    const gx = Math.floor(x / this.cellSize);
    const gy = Math.floor(y / this.cellSize);
    return `${gx},${gy}`;
  }
  
  insert(particle) {
    const key = this.getKey(particle.x, particle.y);
    if (!this.cells.has(key)) {
      this.cells.set(key, []);
    }
    this.cells.get(key).push(particle);
  }
  
  getNeighbors(x, y, radius) {
    const neighbors = [];
    const minGX = Math.floor((x - radius) / this.cellSize);
    const maxGX = Math.floor((x + radius) / this.cellSize);
    const minGY = Math.floor((y - radius) / this.cellSize);
    const maxGY = Math.floor((y + radius) / this.cellSize);
    
    for (let gx = minGX; gx <= maxGX; gx++) {
      for (let gy = minGY; gy <= maxGY; gy++) {
        const key = `${gx},${gy}`;
        const cell = this.cells.get(key);
        if (cell) {
          neighbors.push(...cell);
        }
      }
    }
    
    return neighbors;
  }
}
```

## 性能优化

### 对象池
```javascript
class ParticlePool {
  constructor(size = 10000) {
    this.pool = [];
    this.active = [];
    
    for (let i = 0; i < size; i++) {
      this.pool.push(new Particle());
    }
  }
  
  acquire() {
    let particle = this.pool.pop();
    if (!particle) {
      particle = new Particle();
    }
    this.active.push(particle);
    return particle;
  }
  
  release(particle) {
    const index = this.active.indexOf(particle);
    if (index > -1) {
      this.active.splice(index, 1);
      particle.reset();
      this.pool.push(particle);
    }
  }
  
  update(dt) {
    for (let i = this.active.length - 1; i >= 0; i--) {
      const p = this.active[i];
      p.update(dt);
      
      if (!p.active) {
        this.release(p);
      }
    }
  }
  
  render(ctx) {
    for (const p of this.active) {
      p.render(ctx);
    }
  }
}
```

### 批量绘制
```javascript
function renderParticlesBatch(ctx, particles) {
  // 按颜色分组
  const groups = {};
  
  for (const p of particles) {
    if (!p.active) continue;
    if (!groups[p.color]) groups[p.color] = [];
    groups[p.color].push(p);
  }
  
  // 批量绘制
  for (const [color, group] of Object.entries(groups)) {
    ctx.fillStyle = color;
    ctx.beginPath();
    
    for (const p of group) {
      const progress = p.life / p.maxLife;
      ctx.globalAlpha = 1 - progress;
      ctx.moveTo(p.x + p.size, p.y);
      ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
    }
    
    ctx.fill();
  }
  
  ctx.globalAlpha = 1;
}
```

## 设计原则

1. **生命周期管理**：每个粒子必须有明确的激活/销毁状态
2. **对象池复用**：避免频繁创建/销毁导致 GC
3. **参数化配置**：通过配置而非硬代码创建效果
4. **分层渲染**：静态/动态粒子分离
5. **空间优化**：大量粒子时使用空间分区
6. **批量绘制**：减少状态切换，提高渲染效率
