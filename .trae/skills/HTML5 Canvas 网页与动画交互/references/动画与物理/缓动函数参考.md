# 缓动函数参考

## 什么是缓动函数？

缓动函数（Easing Functions）控制动画速度曲线，让运动更自然、更有生命力。

### 1. 线性（Linear）
```javascript
function linear(t) {
  return t;
}
```
**特点**：匀速运动，机械感强  
**适用**：数据流动、加载进度

### 2. 二次缓动（Quadratic）
```javascript
function easeInQuad(t) {
  return t * t;
}

function easeOutQuad(t) {
  return t * (2 - t);
}

function easeInOutQuad(t) {
  return t < 0.5 ? 2 * t * t : -1 + (4 - 2 * t) * t;
}
```
**特点**：温和的加速/减速  
**适用**：基础 UI 动画、对象移动

### 3. 三次缓动（Cubic）
```javascript
function easeInCubic(t) {
  return t * t * t;
}

function easeOutCubic(t) {
  return (--t) * t * t + 1;
}

function easeInOutCubic(t) {
  return t < 0.5 ? 4 * t * t * t : (t - 1) * (2 * t - 2) * (2 * t - 2) + 1;
}
```
**特点**：比二次更明显的加速/减速  
**适用**：强调动画、页面过渡

### 4. 正弦缓动（Sine）
```javascript
function easeInSine(t) {
  return 1 - Math.cos((t * Math.PI) / 2);
}

function easeOutSine(t) {
  return Math.sin((t * Math.PI) / 2);
}

function easeInOutSine(t) {
  return -(Math.cos(Math.PI * t) - 1) / 2;
}
```
**特点**：最自然的加减速  
**适用**：优雅动画、品牌展示

### 5. 指数缓动（Exponential）
```javascript
function easeInExpo(t) {
  return t === 0 ? 0 : Math.pow(2, 10 * t - 10);
}

function easeOutExpo(t) {
  return t === 1 ? 1 : 1 - Math.pow(2, -10 * t);
}

function easeInOutExpo(t) {
  return t === 0 ? 0 : t === 1 ? 1 : t < 0.5
    ? Math.pow(2, 20 * t - 10) / 2
    : (2 - Math.pow(2, -20 * t + 10)) / 2;
}
```
**特点**：极快开始/结束，戏剧性强  
**适用**：冲击力动画、特效

### 6. 弹性缓动（Elastic）
```javascript
function easeInElastic(t) {
  if (t === 0 || t === 1) return t;
  return -Math.pow(2, 10 * t - 10) * Math.sin((t * 10 - 10.75) * (2 * Math.PI) / 3);
}

function easeOutElastic(t) {
  if (t === 0 || t === 1) return t;
  return Math.pow(2, -10 * t) * Math.sin((t * 10 - 0.75) * (2 * Math.PI) / 3) + 1;
}

function easeInOutElastic(t) {
  if (t === 0 || t === 1) return t;
  return t < 0.5
    ? -(Math.pow(2, 20 * t - 10) * Math.sin((20 * t - 11.125) * (2 * Math.PI) / 4.5)) / 2
    : (Math.pow(2, -20 * t + 10) * Math.sin((20 * t - 11.125) * (2 * Math.PI) / 4.5)) / 2 + 1;
}
```
**特点**：过冲并回弹，弹性感  
**适用**：弹簧效果、交互反馈

### 7. 回弹缓动（Back）
```javascript
function easeInBack(t) {
  const c1 = 1.70158;
  const c3 = c1 + 1;
  return c3 * t * t * t - c1 * t * t;
}

function easeOutBack(t) {
  const c1 = 1.70158;
  const c3 = c1 + 1;
  return 1 + c3 * Math.pow(t - 1, 3) + c1 * Math.pow(t - 1, 2);
}

function easeInOutBack(t) {
  const c1 = 1.70158;
  const c2 = c1 * 1.525;
  return t < 0.5
    ? (Math.pow(2 * t, 2) * ((c2 + 1) * 2 * t - c2)) / 2
    : (Math.pow(2 * t - 2, 2) * ((c2 + 1) * (t * 2 - 2) + c2) + 2) / 2;
}
```
**特点**：轻微过冲，自然感  
**适用**：UI 交互、按钮点击

### 8. 弹跳缓动（Bounce）
```javascript
function easeInBounce(t) {
  return 1 - easeOutBounce(1 - t);
}

function easeOutBounce(t) {
  const n1 = 7.5625;
  const d1 = 2.75;
  
  if (t < 1 / d1) {
    return n1 * t * t;
  } else if (t < 2 / d1) {
    return n1 * (t -= 1.5 / d1) * t + 0.75;
  } else if (t < 2.5 / d1) {
    return n1 * (t -= 2.25 / d1) * t + 0.9375;
  } else {
    return n1 * (t -= 2.625 / d1) * t + 0.984375;
  }
}

function easeInOutBounce(t) {
  return t < 0.5
    ? (1 - easeOutBounce(1 - 2 * t)) / 2
    ? (1 + easeOutBounce(2 * t - 1)) / 2;
}
```
**特点**：多次弹跳，活泼感  
**适用**：游戏、趣味动画

## 使用方式

### 基本插值
```javascript
function lerp(start, end, t) {
  return start + (end - start) * t;
}

function animate(start, end, duration, easingFn) {
  const startTime = performance.now();
  
  function update(currentTime) {
    const elapsed = (currentTime - startTime) / 1000;
    const progress = Math.min(elapsed / duration, 1);
    const easedProgress = easingFn(progress);
    const currentValue = lerp(start, end, easedProgress);
    
    // 使用 currentValue
    render(currentValue);
    
    if (progress < 1) {
      requestAnimationFrame(update);
    }
  }
  
  requestAnimationFrame(update);
}

// 使用
animate(0, 100, 1, easeOutCubic);
```

### 多属性动画
```javascript
function animateObject(obj, targetProps, duration, easingFn) {
  const startTime = performance.now();
  const startProps = {};
  
  for (const key in targetProps) {
    startProps[key] = obj[key];
  }
  
  function update(currentTime) {
    const elapsed = (currentTime - startTime) / 1000;
    const progress = Math.min(elapsed / duration, 1);
    const easedProgress = easingFn(progress);
    
    for (const key in targetProps) {
      obj[key] = lerp(startProps[key], targetProps[key], easedProgress);
    }
    
    if (progress < 1) {
      requestAnimationFrame(update);
    }
  }
  
  requestAnimationFrame(update);
}

// 使用
const particle = { x: 0, y: 0, opacity: 0, scale: 0 };
animateObject(particle, {
  x: 100,
  y: 200,
  opacity: 1,
  scale: 1
}, 0.8, easeOutCubic);
```

## 缓动函数选择指南

| 场景 | 推荐缓动 | 原因 |
|------|---------|------|
| 自然移动 | easeOutSine / easeOutCubic | 柔和减速 |
| UI 交互 | easeOutBack | 轻微过冲 |
| 页面切换 | easeInOutCubic | 平滑过渡 |
| 弹跳球 | easeOutBounce | 物理感 |
| 弹簧效果 | easeOutElastic | 弹性振动 |
| 加载进度 | linear | 匀速可预测 |
| 冲击力特效 | easeOutExpo | 快速爆发 |

## 自定义缓动函数

### 贝塞尔曲线
```javascript
function cubicBezier(p1x, p1y, p2x, p2y) {
  const cx = 3 * p1x;
  const bx = 3 * (p2x - p1x) - cx;
  const ax = 1 - cx - bx;
  
  const cy = 3 * p1y;
  const by = 3 * (p2y - p1y) - cy;
  const ay = 1 - cy - by;
  
  function sampleCurveX(t) {
    return ((ax * t + bx) * t + cx) * t;
  }
  
  function sampleCurveY(t) {
    return ((ay * t + by) * t + cy) * t;
  }
  
  function solveCurveX(x) {
    let t = x;
    for (let i = 0; i < 8; i++) {
      const x2 = sampleCurveX(t) - x;
      if (Math.abs(x2) < 0.001) return t;
      const d2 = (3 * ax * t + 2 * bx) * t + cx;
      if (Math.abs(d2) < 0.000001) break;
      t = t - x2 / d2;
    }
    return sampleCurveY(t);
  }
  
  return function(x) {
    return solveCurveX(x);
  };
}

// CSS 标准缓动对应
const ease = cubicBezier(0.25, 0.1, 0.25, 1);
const easeIn = cubicBezier(0.42, 0, 1, 1);
const easeOut = cubicBezier(0, 0, 0.58, 1);
const easeInOut = cubicBezier(0.42, 0, 0.58, 1);
```

## 性能提示

- 预计算缓动值表（LUT）用于密集场景
- 避免在动画循环中重复创建函数
- 使用简单缓动（Quad/Cubic）性能最佳
- 弹性/弹跳缓计算量较大，适度使用
